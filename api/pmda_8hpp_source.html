281  /**
282  * @brief Run this daemon.
283  *
284  * This function implements the main processing loop of the daemon-mode
285  * PMDA. It performs various initalisations such as parsing the command
286  * line options, then defers processing to the pmdaMain function.
287  *
288  * @param argc Argument count.
289  * @param argv Argument vector.
290  *
291  * @throw pcp::exception On error.
292  */
293  virtual void run_daemon(const int argc, char * const argv[])
294  {
295  // Create some local strings. We keep these as separate variables
296  // (as opposed to using function call results directly) because PMAPI
297  // functions will copy their pointers (not their contents), so the
298  // pointers must remain valid for the life of this function.
299  const std::string program_name = get_pmda_name();
300  const std::string help_text_pathname = get_help_text_pathname();
301  const std::string log_file_pathname = get_log_file_pathname();
302 
303  // Set the program name.
304  __pmSetProgname(program_name.c_str());
305 
306  // Initialize the PMDA to run as a daemon.
307  pmdaInterface interface;
308  pmdaDaemon(&interface, PCP_CPP_PMDA_INTERFACE_VERSION,
309  const_cast<char *>(program_name.c_str()),
310  get_default_pmda_domain_number(),
311  const_cast<char *>(log_file_pathname.c_str()),
312  (help_text_pathname.empty()) ? NULL : const_cast<char *>(help_text_pathname.c_str())
313  );
314 
315  // Parse the command line options.
316  if (!parse_command_line(argc, argv, interface)) {
317  // The parse function already did whatever the command line asked
318  // for (eg --help, or --export-pmns), so we're done now.
319  return;
320  }
321 
322  // Redirect output the log.
323  pmdaOpenLog(&interface);
324 
325  // Initialize the rest of the PMDA.
326  initialize_pmda(interface);
327 
328  // Establish a connection between this daemon PMDA and PMCD.
329  pmdaConnect(&interface);
330 
331  // Run the generic PDU processing loop.
332  pmdaMain(&interface);
333 
334  // Free the instance domains and metrics allocated in initialize_pmda.
335  for (int index = 0; index < interface.version.two.ext->e_nindoms; ++index) {
336  delete [] interface.version.two.ext->e_indoms[index].it_set;
337  }
338  delete[] interface.version.two.ext->e_indoms;
339  delete[] interface.version.two.ext->e_metrics;
340  }

476  get_supported_options().add(get_supported_hidden_options());

481  std::ifstream config_stream(config.as<std::string>().c_str());
482  if (config_stream) {
483  store(parse_config_file(config_stream, supported_options), options);
484  } else if (!config.defaulted()) {
485  throw pcp::exception(PM_ERR_GENERIC, "Failed to open config file: " +
486  config.as<std::string>());
487  }
488 
489 #ifdef PCP_CPP_DEBUG_COMMAND_LINE_OPTIONS
490  for (variables_map::const_iterator iter = options.begin(); iter != options.end(); ++iter) {
491  std::string value;
492  try {
493  if (iter->second.value().type() == typeid(int)) {
494  value = boost::lexical_cast<std::string>(iter->second.as<int>());
495  } else {
496  value = iter->second.as<std::string>();
497  }
498  } catch (const boost::bad_any_cast &ex) {
499  value = "(complex type \"" + std::string(iter->second.value().type().name()) + "\")";

506  if (options.count("help") > 0) {

830  * @brief Initialise this PMDA.
831  *
832  * @param interface PMDA interface to initialise.
833  */
834  virtual void initialize_pmda(pmdaInterface &interface)
835  {
836  // Setup the instance domain and metrics tables. These will be
837  // assigned to members of the interface struct (by pmdaInit), so they
838  // must remain valid as long as the interface does.
839  supported_metrics = get_supported_metrics();
840  const std::pair<size_t,size_t> counts = count_metrics(supported_metrics);
841  const size_t indom_count = counts.second;
842  const size_t metric_count = counts.first;
843  pmdaIndom * indom_table = (indom_count == 0) ? NULL : new pmdaIndom [indom_count];
844  pmdaMetric * metric_table = new pmdaMetric [metric_count];
845 
846  std::map<const instance_domain *, pmInDom> instance_domain_ids;
847  std::vector<instance_domain *> instance_domains;
848  instance_domains.reserve(indom_count);
849  size_t metric_index = 0;
850  for (metrics_description::const_iterator metrics_iter = supported_metrics.begin();
851  metrics_iter != supported_metrics.end(); ++metrics_iter)
852  {
853  const metric_cluster &cluster = metrics_iter->second;
854  for (metric_cluster::const_iterator cluster_iter = cluster.begin();
855  cluster_iter != cluster.end(); ++cluster_iter)
856  {
857  const metric_description &description = cluster_iter->second;
858  assert(metric_index < metric_count);
859  metric_table[metric_index].m_desc = description;
860  metric_table[metric_index].m_desc.pmid =
861  PMDA_PMID(cluster.get_cluster_id(), cluster_iter->first);
862  if (description.domain != NULL) {
863  const std::pair<std::map<const instance_domain *, pmInDom>::const_iterator, bool>
864  insert_result = instance_domain_ids.insert(
865  std::make_pair(cluster_iter->second.domain, instance_domain_ids.size()));
866  const pmInDom indom = insert_result.first->second;
867  assert(indom < indom_count);
868  if (insert_result.second) {
869  indom_table[indom] = allocate_pmda_indom(*cluster_iter->second.domain);
870  instance_domains.push_back(cluster_iter->second.domain);
871  assert(instance_domain_ids.size() == instance_domains.size());
872  }
873  metric_table[metric_index].m_desc.indom = indom;
874  } else {
875  metric_table[metric_index].m_desc.indom = PM_INDOM_NULL;
876  }
877  metric_table[metric_index].m_user = description.opaque;
878  metric_index++;
879  }
880  }
881  assert(instance_domain_ids.size() == indom_count);
882  assert(instance_domains.size() == indom_count);
883  assert(metric_index == metric_count);
884 
885  // Assign our callback function pointers to the interface struct.
886  set_callbacks(interface);
887 
888  // Initialize the PMDA interface.
889  pmdaInit(&interface, indom_table, indom_count, metric_table, metric_count);
890 
891  // Record the pmdaIndom values as updated by pmdaInit.
892  for (size_t indom_index = 0; indom_index < indom_count; ++indom_index) {
893  instance_domain * const domain = instance_domains.at(indom_index);
894  domain->set_pm_instance_domain(indom_table[indom_index].it_indom);
895  this->instance_domains.insert(std::make_pair(domain->get_domain_id(), domain));
896  this->instance_domains.insert(std::make_pair(domain->get_pm_instance_domain(), domain));
897  }
898  }
